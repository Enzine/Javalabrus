33      public FirstUI() throws IOException {
34          initComponents();
35  
36          Timer timer = new Timer();
37          timer.scheduleAtFixedRate(new TimerTask() {
38              @Override
39              public void run() {
40                  pet.hungerDecreasesByOne();
41                  pet.thirstDecreasesByOne();
42                  pet.funDecreasesByOne();
43                  pet.bladderDecreasesByOne();
44                  pet.sleepDecreasesByOne();
45                  try {
46                      mm.recogniseMood(pet);
47                  } catch (FileNotFoundException ex) {
48                      Logger.getLogger(FirstUI.class.getName()).log(Level.SEVERE, null, ex);
49                  }
50                  jTextAreaTraits.setText(pet.toString());
51  
52  //                System.out.print("I would be called every 10 seconds");
53              }
54          }, 2000, 10000);
55          lab = new Lab();
56          pet = new Pet("nemo");
57          try {
58              lab.addPet(pet);
59          } catch (FileNotFoundException ex) {
60              Logger.getLogger(FirstUI.class.getName()).log(Level.SEVERE, null, ex);
61          }
62  
63          jTextAreaTraits.setText(pet.toString());
64  
65          jLabel2.setIcon(new javax.swing.ImageIcon(pet.getPicturePath()));
66          if (!sl.saveFileExists()) {
67              fs.createANewSaveFile("first", "");
68          }
69      }
